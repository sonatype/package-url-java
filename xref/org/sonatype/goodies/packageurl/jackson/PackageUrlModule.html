13  package org.sonatype.goodies.packageurl.jackson;
14  
15  import java.io.IOException;
16  
17  import org.sonatype.goodies.packageurl.PackageUrl;
18  
19  import com.fasterxml.jackson.core.JsonGenerator;
20  import com.fasterxml.jackson.core.JsonParser;
21  import com.fasterxml.jackson.core.JsonProcessingException;
22  import com.fasterxml.jackson.databind.DeserializationContext;
23  import com.fasterxml.jackson.databind.SerializerProvider;
24  import com.fasterxml.jackson.databind.deser.std.StdDeserializer;
25  import com.fasterxml.jackson.databind.module.SimpleModule;
26  import com.fasterxml.jackson.databind.ser.std.StdSerializer;
27  
28  /**
29   * {@link PackageUrl} Jackson module.
30   *
31   * @since ???
32   */
33  public class PackageUrlModule
34      extends SimpleModule
35  {
36    private static final long serialVersionUID = 1L;
37  
38    public PackageUrlModule() {
39      addSerializer(PackageUrl.class, new PackageUrlSerializer());
40      addDeserializer(PackageUrl.class, new PackageUrlDeserializer());
41    }
42  
43    private static class PackageUrlDeserializer
44        extends StdDeserializer<PackageUrl>
45    {
46      private static final long serialVersionUID = 1L;
47  
48      public PackageUrlDeserializer() {
49        super(PackageUrl.class);
50      }
51  
52      @Override
53      public PackageUrl deserialize(final JsonParser parser, final DeserializationContext context)
54          throws IOException, JsonProcessingException
55      {
56        String value = parser.readValueAs(String.class);
57        return PackageUrl.parse(value);
58      }
59    }
60  
61    private static class PackageUrlSerializer
62        extends StdSerializer<PackageUrl>
63    {
64      private static final long serialVersionUID = 1L;
65  
66      public PackageUrlSerializer() {
67        super(PackageUrl.class);
68      }
69  
70      @Override
71      public void serialize(final PackageUrl value, final JsonGenerator generator, final SerializerProvider provider)
72          throws IOException
73      {
74        generator.writeString(value.toString());
75      }
76    }
77  }
