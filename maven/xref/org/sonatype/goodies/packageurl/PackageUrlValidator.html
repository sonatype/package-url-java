13  package org.sonatype.goodies.packageurl;
14  
15  import java.util.List;
16  import java.util.Map;
17  import java.util.regex.Pattern;
18  
19  import javax.annotation.Nullable;
20  
21  /**
22   * {@link PackageUrl} validator.
23   *
24   * @since 1.1.0
25   */
26  class PackageUrlValidator
27  {
28    private PackageUrlValidator() {
29      // empty
30    }
31  
32    private static final Pattern TYPE_PATTERN = Pattern.compile(PackageUrlParser.TYPE);
33  
34    private static final Pattern NAMESPACE_SEGMENT_PATTERN = Pattern.compile(PackageUrlParser.NAMESPACE_SEGMENT);
35  
36    private static final Pattern NAME_PATTERN = Pattern.compile(PackageUrlParser.NAME);
37  
38    private static final Pattern VERSION_PATTERN = Pattern.compile(PackageUrlParser.VERSION);
39  
40    private static final Pattern QUALIFIER_KEY_PATTERN = Pattern.compile(PackageUrlParser.QUALIFIER_KEY);
41  
42    private static final Pattern QUALIFIER_VALUE_PATTERN = Pattern.compile(PackageUrlParser.QUALIFIER_VALUE);
43  
44    private static final Pattern SUBPATH_SEGMENT_PATTERN = Pattern.compile(PackageUrlParser.SUBPATH_SEGMENT);
45  
46    private static void validate(final String type, final Pattern pattern, final String value) {
47      if (!pattern.matcher(value).matches()) {
48        throw new InvalidException("Invalid " + type + ": " + value);
49      }
50    }
51  
52    /**
53     * Validate {@link PackageUrl#type} value.
54     */
55    static void validateType(final String value) {
56      if (value == null) {
57        throw new MissingComponentException("type");
58      }
59      validate("type", TYPE_PATTERN, value);
60    }
61  
62    /**
63     * Validate {@link PackageUrl#namespace} value.
64     */
65    static void validateNamespace(@Nullable final List<String> namespace) {
66      if (namespace != null) {
67        for (String segment : namespace) {
68          validate("namespace.segment", NAMESPACE_SEGMENT_PATTERN, segment);
69        }
70      }
71    }
72  
73    /**
74     * Validate {@link PackageUrl#name} value.
75     */
76    static void validateName(final String value) {
77      if (value == null) {
78        throw new MissingComponentException("name");
79      }
80      // FIXME: name gets url encoding so not sure there is validation to do here?
81      //validate("name", NAME_PATTERN, value);
82    }
83  
84    /**
85     * Validate {@link PackageUrl#version} value.
86     */
87    static void validateVersion(@Nullable final String value) {
88      if (value != null) {
89        validate("version", VERSION_PATTERN, value);
90      }
91    }
92  
93    /**
94     * Validate {@link PackageUrl#qualifiers} value.
95     */
96    static void validateQualifiers(@Nullable final Map<String, String> qualifiers) {
97      if (qualifiers != null) {
98        for (Map.Entry<String, String> entry : qualifiers.entrySet()) {
99          validate("qualifier.key", QUALIFIER_KEY_PATTERN, entry.getKey());
100         // FIXME: value is precent encoding, so not sure there is validation to do here
101         //validate("qualifier.value", QUALIFIER_VALUE_PATTERN, entry.getValue());
102       }
103     }
104   }
105 
106   /**
107    * Validate {@link PackageUrl#subpath} value.
108    */
109   static void validateSubpath(@Nullable final List<String> subpath) {
110     if (subpath != null) {
111       for (String segment : subpath) {
112         validate("subpath.segment", SUBPATH_SEGMENT_PATTERN, segment);
113       }
114     }
115   }
116 }
