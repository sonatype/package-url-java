13  package org.sonatype.goodies.packageurl;
14  
15  import javax.annotation.Nullable;
16  
17  /**
18   * How to render the Package-URL.
19   *
20   * @since 1.1.0
21   */
22  public enum RenderFlavor
23  {
24    /**
25     * Render with {@link PackageUrl#SCHEME} (default).
26     */
27    SCHEME,
28  
29    /**
30     * Render with-out {@link PackageUrl#SCHEME}.
31     */
32    SCHEMELESS;
33  
34    private static volatile RenderFlavor _default = SCHEME;
35  
36    public static RenderFlavor getDefault() {
37      return _default;
38    }
39  
40    /**
41     * Set the default flavor, or null for the original default.
42     */
43    public static void setDefault(@Nullable final RenderFlavor flavor) {
44      _default = flavor == null ? SCHEME : flavor;
45    }
46  }
